platform :ios, '13.0'

# Use static linking instead of dynamic frameworks to avoid Swift issues
use_frameworks! :linkage => :static 

target 'Runner' do
  flutter_install_all_ios_pods File.dirname(File.realpath(__FILE__))

  post_install do |installer|
    installer.pods_project.targets.each do |target|
      target.build_configurations.each do |config|
        config.build_settings['IPHONEOS_DEPLOYMENT_TARGET'] = '13.0'
        config.build_settings['BUILD_LIBRARY_FOR_DISTRIBUTION'] = 'YES'  # Ensures module compatibility
        config.build_settings['SWIFT_INCLUDE_PATHS'] = '$(SRCROOT)/Pods/Headers/Public' # Fix for bridging headers
      end
    end
  end
end
